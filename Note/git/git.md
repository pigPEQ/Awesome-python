```
# 初始化
git init
# 提交到暂存区
git add .
# 删除暂存区的文件
git rm [file]
# 查看暂存区记录
git status
# 提交至本地仓库
git commit -m 'v1.0'
# 文件撤销至工作区
git reset HEAD [file]
# 文件回退
git checkout [file]
# 标签
git tag 
# 查看分支
git branch
# 创建分支
git branch -b [dev]
# 切换分支
git checkout [dev]
# 合并分支（在主分支中）
git merge [dev]
# 将本地分支推送远程
git push -u origin [dev]
# 推送到远程
git push 
# 拉取到本地
git pull
# 查看历史
git log
# 版本回退
git reset --hard HEAD^
git reset --hard 版本号
```
